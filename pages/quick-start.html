Create Nuek Component in your main.js or component-helper script with:

const component = new NuekComponent("CSS Selector", "filePath", propsObject, numberOfLoop, condition);

Then, create a .nuek file with the following structure:

                
<script>
    // JavaScript logic
</script>

<component>
    <!-- Your HTML here -->
</component>

<style>
    /* Component-specific CSS */
</style>

# Using the NPM Package

You can also use NuekJS as an installed npm package for better compatibility with other tools. Just run the command below:

npm install nuekjs

You can integrate NuekJS into various modern JavaScript frameworks using the NPM package. Select a framework below to see the setup example:


import React, { useEffect } from 'react'
import { NuekComponent } from 'nuekjs'

function App() {
  useEffect(() => {
    new NuekComponent('#nuek-container', 'components/Card.nuek', {
      title: 'Hello React'
    })
  }, [])

  return <div id="nuek-container" />
}

export default App

import { Component, AfterViewInit } from '@angular/core'
import { NuekComponent } from 'nuekjs'

@Component({
  selector: 'app-nuek',
  template: '<div id="angular-nuek"></div>'
})
export class NuekComponentWrapper implements AfterViewInit {
  ngAfterViewInit() {
    new NuekComponent('#angular-nuek', 'components/Card.nuek', {
      title: 'Hello Angular'
    })
  }
}
